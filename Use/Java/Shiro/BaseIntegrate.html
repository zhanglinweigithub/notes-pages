                              一、Shiro提供的过滤器

                              详情
                              public enum DefaultFilter {
    // 匿名拦截器，即不需要登录即可访问
    anon(AnonymousFilter.class),
    
    // 必须登录认证才能访问（remember me也不行，必须登录）
    authc(FormAuthenticationFilter.class),
    
    authcBasic(BasicHttpAuthenticationFilter.class),
    
    authcBearer(BearerHttpAuthenticationFilter.class),
    
    // 退出拦截器，主要属性：redirectUrl：退出成功后重定向的地址
    logout(LogoutFilter.class),
    
    noSessionCreation(NoSessionCreationFilter.class),
    
    perms(PermissionsAuthorizationFilter.class),
    
    port(PortFilter.class),
    
    rest(HttpMethodPermissionFilter.class),
    
    roles(RolesAuthorizationFilter.class),
    
    ssl(SslFilter.class),
    
    // 用户拦截器，用户已经身份验证/记住我登录的都可
    user(UserFilter.class),
    
    invalidRequest(InvalidRequestFilter.class);
    // ...
}
                              • 我们无需关系其它，只需要自定义一个拦截器，所有需要登录访问的都走我们自定义的拦截器即可
                              • 其余无需登录直接使用anon即可

                              二、整合

                              整体步骤如下：

                              1. .yaml配置文件定义路径拦截规则
                              2. 实现AuthenticationToken，重写账号密码返回方法
                              3. 继承AuthenticatingFilter自定义拦截逻辑
                              4. 继承AuthorizingRealm，自定义认证、授权逻辑
                              5. 创建ShiroProperties类，接收配置文件内容
                              6. 创建Shiro配置类，注入我们自己的过滤器和路径拦截规则（有需要的也可以自定义Cookie、SessionManager等）

@Data
@Builder
public class AuthToken implements AuthenticationToken {

    private String userName;

    private String password;

    private SysUser loginUser;

    @Override
    public Object getPrincipal() {
        return this.userName;
    }

    @Override

                              AuthFilter
                              /**
 * 自定义auth过滤器
 */
public class AuthFilter extends AuthenticatingFilter {

    private final String TOKEN_HEADER_KEY = "Authorization";

    /**
     * 创建登录Token（在调用executeLogin时才会调用，基本没啥用）
     */
    @Override
    protected AuthenticationToken createToken(ServletRequest request, ServletResponse response) throws Exception {
        return null;
    }

    /**
     * 访问被拒绝时
     */
    @Override
    protected boolean onAccessDenied(ServletRequest request, ServletResponse response) throws Exception {
        SysUser principal = (SysUser) SecurityUtils.getSubject().getPrincipal();
        if (principal != null) {
            SecurityUtils.getSubject().logout();
        }

        ((HttpServletResponse) response).sendError(401, "未登录");

        return false;
    }


    /**
     * 是否允许访问
     */
    @Override
    protected boolean isAccessAllowed(ServletRequest request, ServletResponse response, Object mappedValue) {
        if (((HttpServletRequest) request).getMethod().equals(RequestMethod.OPTIONS.name())) {
            return true;
        }
        // shiro默认使用Cookie+session判断是否登录, 若想纯靠header判断, 可删除此行
        boolean accessAllowed = super.isAccessAllowed(request, response, mappedValue);
        if (accessAllowed){
            String token = ((HttpServletRequest) request).getHeader(TOKEN_HEADER_KEY);
            return StringUtils.isNoneBlank(token) && LoginUtil.isLogin(token);
        }
        return false;
    }

}
                              LoginRealm
                              /**
 * 自定义login认证
 */
public class LoginRealm extends AuthorizingRealm {

    @Autowired
    private SysUserMapper sysUserMapper;

    /**
     * 授权
     * @param principalCollection
     * @return
     */
    @Override
    protected AuthorizationInfo doGetAuthorizationInfo(PrincipalCollection principalCollection) {
        // 如果你有多个Realm，这里需要注意，后面再说
        SysUser principal = (SysUser) principalCollection.getPrimaryPrincipal();
        SimpleAuthorizationInfo info = new SimpleAuthorizationInfo();
        // 这里的角色和权限可以从数据库中查询
        info.addRoles("admin");
        info.addStringPermissions("user:delete");
        
        // 或者在认证时直接把角色、权限信息查出来，这里直接设置就不用查询数据库了
        // SysUser userPrincipal = (SysUser) SecurityUtils.getSubject().getPrincipal();
        // info.addRoles(userPrincipal.getRoleCodes());
        // info.addStringPermissions(userPrincipal.getPermissionsCode());
        return info;
    }

    /**
     * 认证
     * @param authenticationToken
     * @return
     * @throws AuthenticationException
     */
    @Override
    protected AuthenticationInfo doGetAuthenticationInfo(AuthenticationToken authenticationToken) throws AuthenticationException {
        if (authenticationToken instanceof AuthToken) {
            String userName = ((AuthToken) authenticationToken).getUserName();
            SysUser user = sysUserMapper.findByUserName(userName);
            if (user != null) {
                // 用于多个Realm时，单个Realm也能用，不影响，后面再说
                final PrincipalCollection principalCollection = new SimplePrincipalCollection(user, getName());
            return new SimpleAuthenticationInfo(principalCollection,user.getPassword());
            }
        }
        return null;
    }

    /**
     * 判断token类型是否允许
     */
    @Override
    public boolean supports(AuthenticationToken token) {
        return token instanceof AuthToken;
    }
}
                              ShiroProperties
                              /**
 * 读取shiro的相关配置内容
 */
@Data
public class ShiroProperties {

    /**
     * 路径拦截规则
     */
    private List<String> filterChainDefinitions;

    /**
     * 转成 key, value
     * key: 路径， value: 拦截规则
     * @return
     */
    public Map<String, String> filterChainDefinitionsToMap() {
        Map<String, String> map = new LinkedHashMap<>();
        this.filterChainDefinitions.forEach(item -> {
            String[] split = item.split(":");
            map.put(split[0], split[1]);
        });
        return map;
    }

}
                              ShiroConfig
                              @Configuration
public class ShiroConfig {

    @Autowired
    private ShiroProperties shiroProperties;

    /**
     * 配置SecurityManager
     */
    @Bean
    public DefaultWebSecurityManager defaultWebSecurityManager(LoginRealm loginRealm){
        DefaultWebSecurityManager defaultWebSecurityManager = new DefaultWebSecurityManager();
        defaultWebSecurityManager.setRealm(loginRealm);
        return defaultWebSecurityManager;
    }

    /**
     * 注入Realm
     */
    @Bean
    public LoginRealm loginRealm(){
        return new LoginRealm();
    }

    /**
     * 配置过滤器、拦截范围
     */
    @Bean
    public ShiroFilterFactoryBean shiroFilterFactoryBean(DefaultWebSecurityManager securityManager){
        ShiroFilterFactoryBean shiroFilterFactoryBean = new ShiroFilterFactoryBean();
        shiroFilterFactoryBean.setSecurityManager(securityManager);
        // 设置自定义 auth 过滤器
        Map<String, Filter> filters = new HashMap<>();
        filters.put("auth", new AuthFilter());
        shiroFilterFactoryBean.setFilters(filters);
        // 设置路径拦截规则
        Map<String, String> filterMap = new LinkedHashMap<>(shiroProperties.filterChainDefinitionsToMap());
        shiroFilterFactoryBean.setFilterChainDefinitionMap(filterMap);
        return shiroFilterFactoryBean;
    }

}

                              三、流程说明

                              1、发起请求时首先根据配置文件中的路径拦截规则进行拦截

                              2、当执行登录时，首先根据用户输入内容创建Token

                              public String login(SysUser req) {
    AuthToken authToken = AuthToken.builder()
        .userName(req.getUserName())
        .password(req.getPassword())
        .build();
    SecurityUtils.getSubject().login(authToken);
    String token = UUID.randomUUID().toString().replaceAll("-", "");
    LoginUtil.add(token);
    return token;
}
                              • 当执行SecurityUtils.getSubject().login(authToken);时，会携带我们创建的Token调用自定义Realm进行认证
                              • 认证失败会抛出异常，认证成功会将用户信息保存
                              // user就是保存的用户信息，接收Object类型
final PrincipalCollection principalCollection = new SimplePrincipalCollection(user, getName());
            return new SimpleAuthenticationInfo(principalCollection,user.getPassword());
                              • 调用SecurityUtils.getSubject().getPrincipal();获取的就是我们上面保存的信息，根据Session区分，SessionId保存在Cookie中

                              登出

                              登出时直接调用SecurityUtils.getSubject().logout();，然后清理一下自己保存在Redis之类的信息即可

                              四、多Realm配置

                              修改ShiroConfig类，添加ModularRealmAuthenticator 对象

                              @Configuration
public class ShiroConfig {

    @Autowired
    private ShiroProperties shiroProperties;

    /**
     * 配置SecurityManager
     */
    @Bean
    public DefaultWebSecurityManager defaultWebSecurityManager(LoginRealm loginRealm){
        DefaultWebSecurityManager defaultWebSecurityManager = new DefaultWebSecurityManager();
        // 存入认证对象
        defaultWebSecurityManager.setAuthenticator(modularRealmAuthenticator());
        defaultWebSecurityManager.setRealm(loginRealm);
        return defaultWebSecurityManager;
    }
    
    /**
     * 配置多Realm策略
     */
    public ModularRealmAuthenticator modularRealmAuthenticator() {
        ModularRealmAuthenticator modularRealmAuthenticator = new ModularRealmAuthenticator();
        modularRealmAuthenticator.setAuthenticationStrategy(new AllSuccessfulStrategy());
        return modularRealmAuthenticator;
    }

    /**
     * 注入Realm
     */
    @Bean
    public LoginRealm loginRealm(){
        return new LoginRealm();
    }

    /**
     * 配置过滤器、拦截范围
     */
    @Bean
    public ShiroFilterFactoryBean shiroFilterFactoryBean(DefaultWebSecurityManager securityManager){
        ShiroFilterFactoryBean shiroFilterFactoryBean = new ShiroFilterFactoryBean();
        shiroFilterFactoryBean.setSecurityManager(securityManager);
        // 设置自定义 auth 过滤器
        Map<String, Filter> filters = new HashMap<>();
        filters.put("auth", new AuthFilter());
        shiroFilterFactoryBean.setFilters(filters);
        // 设置路径拦截规则
        Map<String, String> filterMap = new LinkedHashMap<>(shiroProperties.filterChainDefinitionsToMap());
        shiroFilterFactoryBean.setFilterChainDefinitionMap(filterMap);
        return shiroFilterFactoryBean;
    }

}
                              • AtLeastOneSuccessfulStrategy：只要有一个（或更多）的 Realm 验证成功，那么认证将视为成功（默认）
                              • FirstSuccessfulStrategy：第一个 Realm 验证成功，整体认证将视为成功，且后续 Realm 将被忽略
                              • AllSuccessfulStrategy：所有 Realm 成功，认证才视为成功

                              五、PrincipalCollection

                              用于多Realm情况

                              当我们有多个自定义Realm时，因为每个Realm中都保存了一个当前用户信息，通过该接口可指定获取哪个Realm存的信息，默认返回第一条，但是其内部是Map存储，也就相当于是随机返回一个了

                              六、其它说明

                              默认情况下：

                              • 如果要在过滤器中获得请求ID，需要使用String requestedSessionId = ((ShiroHttpServletRequest) request).getRequestedSessionId();获取，否则获取到的SessionId与Cookie关联的不同

                              • Shiro使用Session + Cookie 判断当前登录用户，Session有效期默认30分

                              • Shiro会在每次请求时调用touch()方法刷新Session有效期

                              • 可以调用stop()方法强制清除Session（踢下线）

                              // false：表示获取不到返回null
// true：获取不到创建信的session
Session session = SecurityUtils.getSubject().getSession(false);
if (session != null) {
    session.stop(); // 踢下线，logout时会自动调用
    session.touch(); // 刷新session有效期，每次请求自动调用
}
