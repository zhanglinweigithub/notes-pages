              @ParamLog
              /**
 * 打印日志
 */
@Target({ElementType.METHOD, ElementType.TYPE})
@Retention(RetentionPolicy.RUNTIME)
public @interface ParamLog {

    boolean show() default true;

}
              ParamLogAspect
              /**
 * @ParamLog 注解增强
 */
@Aspect
@Component
@Slf4j
public class ParamLogAspect {

    @Pointcut("@within(com.zhang.vueAdmin.annotation.ParamLog) || @annotation(com.zhang.vueAdmin.annotation.ParamLog)")
    public void aspectLog() {

    }

    /**
     * 环绕通知
     * @param point
     * @return
     * @throws Throwable
     */
    @Around("aspectLog()")
    public Object around(ProceedingJoinPoint point) throws Throwable {
        // 获得方法签名
        MethodSignature methodSignature = (MethodSignature) point.getSignature();
        // 判断是否需要打印日志
        if (Boolean.FALSE.equals(needLog(point, methodSignature))) {
            return point.proceed();
        }
        LocalDateTime start = LocalDateTime.now();
        Object value = point.proceed();
        LocalDateTime end = LocalDateTime.now();
        try {
            log.info("==========================start=============================");
            // 打印日志
            this.basicLogMap(point, methodSignature, Level.INFO);
            // 计算请求耗时
            long diffTime = end.toInstant(ZoneOffset.UTC).toEpochMilli() - start.toInstant(ZoneOffset.UTC).toEpochMilli();
            log.info("useTime: {}", diffTime);
            if(Objects.nonNull(value)){
                log.info("response: {}", JSON.toJSONString(value));
            }
            log.info("==========================end===============================");

        } catch (Exception e) {
            log.error("parse json error", e);
        }

        return value;
    }

    /**
     * 异常通知
     * @param point
     * @param exception
     */
    @AfterThrowing(value = "aspectLog()", throwing = "exception")
    public void throwing(JoinPoint point, Throwable exception) {
        // 拿到方法签名
        MethodSignature methodSignature = (MethodSignature) point.getSignature();
        // 判断是否需要打印日志
        if (Boolean.FALSE.equals(needLog(point, methodSignature))) {
            return;
        }
        // 打印日志
        this.basicLogMap(point, methodSignature, Level.ERROR);
        this.log(Level.ERROR,"exception: {}", ExceptionUtils.getMessage(exception));
    }


    /**
     * 打印日志
     * @param point
     * @param methodSignature
     * @param level
     */
    private void basicLogMap(JoinPoint point, MethodSignature methodSignature, Level level) {
        ServletRequestAttributes attributes = (ServletRequestAttributes) RequestContextHolder.getRequestAttributes();
        if (Objects.nonNull(attributes)) {
            log(level, "url: {}", attributes.getRequest().getRequestURL().toString());
        }
        this.log(level, "className: {}", point.getTarget().getClass().getName());
        this.log(level, "method: {}", methodSignature.getMethod().getName());
        this.log(level, "params: {}", JSON.toJSONString(toParamMap(point)));

    }

    /**
     * 真正打印对应级别得日志
     * @param level
     * @param content
     * @param args
     */
    private void log(Level level, String content, Object... args) {
        if (level == Level.ERROR) {
            log.error(content, args);
        } else if (level == Level.INFO) {
            log.info(content, args);
        } else if(level==Level.WARN){
            log.warn(content, args);
        }else {
            log.debug(content, args);
        }
    }

    /**
     * 判断是否需要打印日志
     * @param point
     * @param methodSignature
     * @return
     */
    private boolean needLog(JoinPoint point, MethodSignature methodSignature) {
        // 拿到方法上得注解
        ParamLog paramLog = methodSignature.getMethod().getAnnotation(ParamLog.class);
        if (Objects.isNull(paramLog)) {
            // 方法上没有注解则取类上得注解
            paramLog = point.getTarget().getClass().getAnnotation(ParamLog.class);
        }
        return Objects.nonNull(paramLog) && paramLog.show();
    }


    /**
     * 处理数据
     * @param joinPoint
     * @return
     */
    private Map toParamMap(JoinPoint joinPoint) {
        Object[] paramValues = joinPoint.getArgs();
        String[] paramNames = ((CodeSignature) joinPoint.getSignature()).getParameterNames();
        Map<String, Object> resultMap = new HashMap<>();
        for (int i = 0; i < paramNames.length; i++) {
            if (!isHttpParam(paramValues[i])) {
                //非http相关参数，才进行打印数据
                resultMap.put(paramNames[i], paramValues[i]);
            }
        }
        return resultMap;
    }

    /**
     * 判断是否 HTTP 参数
     * @param object
     * @return
     */
    private boolean isHttpParam(Object object) {
        return (object instanceof HttpServletRequest) ||
                (object instanceof HttpServletResponse) ||
                (object instanceof HttpSession) ||
                (object instanceof MultipartFile);
    }

}

@Retention(RetentionPolicy.RUNTIME)

     * 加锁过期时间，默认是5秒
     * @return
     */
    long lockTime() default 5L;
}
              RepeatSubmitAspect
              @Component
@Slf4j
@Aspect
public class RepeatSubmitAspect {

    @Resource
    private RedisTemplate redisTemplate;

    @Pointcut("@annotation(repeatSubmit)")
    public void pointCutNoRepeatSubmitAspect(RepeatSubmit repeatSubmit) {

    }

    @Around("pointCutNoRepeatSubmitAspect(repeatSubmit)")
    public Object around(ProceedingJoinPoint joinPoint,RepeatSubmit repeatSubmit) throws Throwable {
        String reqSignature = buildReqSignature(joinPoint);
        //加锁时间
        long lockTime = repeatSubmit.lockTime();
        Boolean res = redisTemplate.opsForValue().setIfAbsent(reqSignature, 1, lockTime, TimeUnit.SECONDS);
        if(!res){
            throw new RuntimeException("重复请求！");
        }
        return joinPoint.proceed();
    }

    /**
     * 构建请求签名
     * @param joinPoint
     * @return
     */
    private String buildReqSignature(ProceedingJoinPoint joinPoint) {
        ServletRequestAttributes requestAttributes = (ServletRequestAttributes) RequestContextHolder.getRequestAttributes();
        if(Objects.isNull(requestAttributes)){
            throw new RuntimeException("请求参数异常！");
        }
        HttpServletRequest request = requestAttributes.getRequest();
        String remoteAddr =request.getRemoteAddr()  + ":" +  request.getRemoteHost();
        String method = request.getMethod();
        String requestURI = request.getRequestURI();
        StringBuffer sb = new StringBuffer();
        for (Object arg : joinPoint.getArgs()) {
            sb.append(arg);
        }
        //请求签名(MD5加密)
        return DigestUtil.md5Hex(StrUtil.format("{}-{}-{}-{}",remoteAddr,method,requestURI,sb));
    }
}

              public class StreamUtils {
    private StreamUtils(){}

    public static <K, V> Map<K, V> toMap(Collection<V> list, Function<V, K> function) {
        return list.stream()
                .collect(Collectors.toMap(function, Function.identity()));
    }

    public static <K, V, T> Map<K, T> toMap(Collection<V> list, Function<V, K> keyMapper, Function<V, T> valueMapper) {
        return list.stream()

    }

    public static <K, V> Map<K, V> toMergedMap(Collection<V> list, Function<V, K> function) {
        return list.stream().collect(Collectors.toMap(function, Function.identity(), (a, b) -> a));
    }

    public static <K, V> Map<K, V> toMergedMap(List<V> list, Function<V, K> function, BinaryOperator<V> operator) {
        return list.stream().collect(Collectors.toMap(function, Function.identity(), operator));
    }

    public static <K, V, T> Map<K, T> toMergedMap(Collection<V> list, Function<V, K> keyMapper, Function<V, T> valueMapper) {
        return list.stream()
                .collect(Collectors.toMap(keyMapper, valueMapper, (a, b) -> a));
    }

    public static <K, V> Map<K, List<V>> group(List<V> list, Function<V, K> keyMapper) {
        return group(list, keyMapper, Function.identity());
    }

    public static <T, K, V> Map<K, List<V>> group(List<T> list, Function<T, K> keyMapper, Function<T, V> valueMapper) {
        return list.stream()
                .collect(
                        Collectors.groupingBy(
                                keyMapper, Collectors.mapping(
                                        valueMapper, Collectors.toList())));
    }

    public static <T, K, V> Map<K, Set<V>> uniqueGroup(List<T> list, Function<T, K> keyMapper, Function<T, V> valueMapper) {
        return list.stream()
                .collect(
                        Collectors.groupingBy(
                                keyMapper, Collectors.mapping(
                                        valueMapper, Collectors.toSet())));
    }

    public static <S, T> List<T> toList(Collection<S> list, Function<S, T> function) {

    public static <S, T> Set<T> findDuplicates(Collection<S> list, Function<S, T> valueMapper) {
        // 从这里可以看出,在hashmap中将一个value remove时,对应的key也会被remove掉
        return list.stream()
                .collect(Collectors.collectingAndThen(groupingBy(valueMapper, counting()),
                        map -> {
                            map.values().removeIf(v -> v < 2);
                            return map.keySet();
                        }));
    }

    public static <T> String join(Collection<T> list, String separator) {
        return list.stream().map(T::toString).collect(Collectors.joining(separator));
    }

    public static <T> String join(Collection<T> list, Function<T, String> valueMapper, String separator) {
        return list.stream().map(valueMapper).collect(Collectors.joining(separator));
    }

    /**
     * join前移除重复的元素
     */
    public static <T> String joinUnique(Collection<T> list, Function<T, String> valueMapper, String separator) {
        Set<String> set = new HashSet<>();
        return list.stream().map(valueMapper).filter(set::add).collect(Collectors.joining(separator));
    }

    /**
     * 将一个数组转换为set
     */
    @SafeVarargs
    public static <T> Set<T> toSet(T... ts) {
        return Arrays.stream(ts).collect(Collectors.toSet());
    }

    @SafeVarargs
    public static <T> void forEach(Collection<T> ts, Consumer<T>... consumers) {
        ts.forEach(t -> {
            for (Consumer<T> consumer : consumers) {
                consumer.accept(t);
            }
        });
    }
